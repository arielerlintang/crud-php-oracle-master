
MEMBUAT TABLE BARU
-----------------
CREATE TABLE SISWA (
  ID_SISWA NUMBER(11) PRIMARY KEY,
  NAMA_SISWA VARCHAR2(100),
  ALAMAT_SISWA CLOB,
  JK_SISWA VARCHAR2(10) CHECK (JK_SISWA IN ('Laki-laki', 'Perempuan')),
  FOTO_SISWA VARCHAR2(100)
);
